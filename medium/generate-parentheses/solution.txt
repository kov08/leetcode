    private void backtrack(List<String> result, StringBuilder current, int openCount, int 
    closeCount, int maxPairs) {
        // Base case: If the current string length reaches 2 * n, a valid combination is formed
        if (current.length() == maxPairs * 2) {
            result.add(current.toString());
            return;
        }
    public List<String> generateParenthesis(int n) {
        List<String> result = new ArrayList<>();

        // Rule 1: We can add an opening parenthesis if we haven't placed all 'n' of them yet
        if (openCount < maxPairs) {
            current.append('(');
            backtrack(result, current, openCount + 1, closeCount, maxPairs);
            current.deleteCharAt(current.length() - 1); // Backtrack step
        }

        // Rule 2: We can add a closing parenthesis only if there are more open than close ones
        if (closeCount < openCount) {
            current.append(')');
            backtrack(result, current, openCount, closeCount + 1, maxPairs);
            current.deleteCharAt(current.length() - 1); // Backtrack step
        }
    }
